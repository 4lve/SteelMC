19#[derive(Debug, Clone, Copy, PartialEq, Eq)]
20pub enum InteractionResult {
21    Success,
22    Fail,
23    Pass,
24    TryEmptyHandInteraction,
25}
26
27impl InteractionResult {
28    /// Returns true if this result consumes the action (Success or Fail).
29    /// Pass and TryEmptyHandInteraction do not consume the action.
30    #[must_use]
31    pub fn consumes_action(&self) -> bool {
32        matches!(self, InteractionResult::Success | InteractionResult::Fail)
33    }
34}

36#[derive(Debug, Clone)]
37pub struct BlockHitResult {
38    pub location: Vector3<f64>,
39    pub direction: Direction,
40    pub block_pos: BlockPos,
41    pub miss: bool,
42    pub inside: bool,
43    pub world_border_hit: bool,
44}
45
46impl ReadFrom for BlockHitResult {
47    fn read(data: &mut impl Read) -> io::Result<Self> {
48        let block_pos = BlockPos::read(data)?;
49        let direction = Direction::read(data)?;
50        // Click coordinates are relative to the block position (0.0 to 1.0 range)
51        let click_x = f32::read(data)?;
52        let click_y = f32::read(data)?;
53        let click_z = f32::read(data)?;
54        let inside = bool::read(data)?;
55        let world_border_hit = bool::read(data)?;
56
57        // Convert to absolute world coordinates by adding block position
58        // (matching Java's FriendlyByteBuf.readBlockHitResult)
59        let location = Vector3::new(
60            f64::from(block_pos.x()) + f64::from(click_x),
61            f64::from(block_pos.y()) + f64::from(click_y),
62            f64::from(block_pos.z()) + f64::from(click_z),
63        );
64
65        Ok(BlockHitResult {
66            location,
67            direction,
68            block_pos,
69            miss: false,
70            inside,
71            world_border_hit,
72        })
73    }
74}

98/// Behavior for items that place blocks
99pub struct BlockItemBehavior {
100    pub block: BlockRef,
101}
102
103impl ItemBehavior for BlockItemBehavior {
104    fn use_on(&self, context: &mut UseOnContext) -> InteractionResult {
105        let clicked_pos = context.hit_result.block_pos;
106        let clicked_state = context.world.get_block_state(&clicked_pos);
107
108        // Get the clicked block to check if it's replaceable
109        let clicked_block = REGISTRY.blocks.by_state_id(clicked_state);
110        let clicked_replaceable = clicked_block.map(|b| b.config.replaceable).unwrap_or(false);
111
112        // Determine placement position: replace clicked block if replaceable,
113        // otherwise place adjacent to the clicked face
114        let (place_pos, replace_clicked) = if clicked_replaceable {
115            (clicked_pos, true)
116        } else {
117            (context.hit_result.direction.relative(&clicked_pos), false)
118        };
119
120        // Check if placement position is within world bounds
121        if !context.world.is_in_valid_bounds(&place_pos) {
122            return InteractionResult::Fail;
123        }
124
125        // Check if the placement position already has a non-replaceable block
126        let existing_state = context.world.get_block_state(&place_pos);
127        let existing_block = REGISTRY.blocks.by_state_id(existing_state);
128        let existing_replaceable = existing_block
129            .map(|b| b.config.replaceable)
130            .unwrap_or(false);
131
132        if !existing_replaceable {
133            return InteractionResult::Fail;
134        }
135
136        // Create placement context
137        // TODO: Calculate horizontal_direction from player rotation
138        let place_context = BlockPlaceContext {
139            clicked_pos,
140            clicked_face: context.hit_result.direction,
141            click_location: context.hit_result.location,
142            inside: context.hit_result.inside,
143            relative_pos: place_pos,
144            replace_clicked,
145            horizontal_direction: Direction::North, // TODO: Calculate from player rotation
146            rotation: 0.0,                          // TODO: Get from player
147            world: context.world,
148        };
149
150        // Get block state for placement from the block behavior
151        let behavior = REGISTRY.blocks.get_behavior(self.block);
152        let Some(new_state) = behavior.get_state_for_placement(&place_context) else {
153            return InteractionResult::Fail;
154        };
155
156        // Place the block
157        if !context
158            .world
159            .set_block(place_pos, new_state, UpdateFlags::all())
160        {
161            return InteractionResult::Fail;
162        }
163
164        // Consume one item from the stack (like Java's itemStack.consume(1, player))
165        context.item_stack.shrink(1);
166
167        // TODO: Play place sound
168        // TODO: Call behavior.on_place()
169
170        InteractionResult::Success
171    }
172}
