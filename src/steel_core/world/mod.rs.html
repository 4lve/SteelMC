60    /// Creates a new world.
61    ///
62    /// Uses `Arc::new_cyclic` to create a cyclic reference between
63    /// the World and its `ChunkMap`'s `WorldGenContext`.
64    #[allow(clippy::new_without_default)]
65    pub async fn new(
66        chunk_runtime: Arc<Runtime>,
67        dimension: DimensionTypeRef,
68        seed: i64,
69    ) -> io::Result<Arc<Self>> {
70        let level_data =
71            LevelDataManager::new(format!("world/{}", dimension.key.path), seed).await?;
72
73        Ok(Arc::new_cyclic(|weak_self: &Weak<World>| Self {
74            chunk_map: Arc::new(ChunkMap::new(chunk_runtime, weak_self.clone(), &dimension)),
75            players: PlayerMap::new(),
76            player_area_map: PlayerAreaMap::new(),
77            dimension,
78            level_data: SyncRwLock::new(level_data),
79            tick_runs_normally: AtomicBool::new(true),
80        }))
81    }

83    /// Cleans up the world by saving all chunks.
84    /// `await_holding_lock` is safe here cause it's only done on shutdown
85    #[allow(clippy::await_holding_lock)]
86    pub async fn cleanup(&self, total_saved: &mut usize) {
87        match self.level_data.write().save_force().await {
88            Ok(()) => log::info!(
89                "World {} level data saved successfully",
90                self.dimension.key.path
91            ),
92            Err(e) => log::error!("Failed to save world level data: {e}"),
93        }
94
95        match self.save_all_chunks().await {
96            Ok(count) => *total_saved += count,
97            Err(e) => log::error!("Failed to save world chunks: {e}"),
98        }
99    }

190    /// Gets the obfuscated seed for sending to clients.
191    ///
192    /// This uses SHA-256 hashing to prevent clients from easily extracting
193    /// the actual world seed, matching vanilla's `BiomeManager.obfuscateSeed()`.
194    #[must_use]
195    #[allow(clippy::missing_panics_doc)] // SHA-256 always produces 32 bytes
196    pub fn obfuscated_seed(&self) -> i64 {
197        let seed = self.seed();
198        let mut hasher = Sha256::new();
199        hasher.update(seed.to_be_bytes());
200        let result = hasher.finalize();
201        // SHA-256 always produces 32 bytes, so taking 8 bytes always succeeds
202        let bytes: [u8; 8] = result[0..8].try_into().expect("SHA-256 produces 32 bytes");
203        i64::from_be_bytes(bytes)
204    }

206    /// Gets the block state at the given position.
207    ///
208    /// Returns the default block state (void air) if the position is out of bounds or the chunk is not loaded.
209    #[must_use]
210    pub fn get_block_state(&self, pos: &BlockPos) -> BlockStateId {
211        if !self.is_in_valid_bounds(pos) {
212            return REGISTRY.blocks.get_base_state_id(vanilla_blocks::AIR);
213        }
214
215        let Some(chunk) = self.get_chunk_at(pos) else {
216            return REGISTRY.blocks.get_base_state_id(vanilla_blocks::AIR);
217        };
218
219        chunk.get_block_state(*pos)
220    }

230    /// Sets a block at the given position with a custom update limit.
231    ///
232    /// The update limit prevents infinite recursion when shape updates trigger
233    /// further block changes. Each recursive call decrements the limit.
234    ///
235    /// Returns `true` if the block was successfully set, `false` otherwise.
236    pub fn set_block_with_limit(
237        &self,
238        pos: BlockPos,
239        block_state: BlockStateId,
240        flags: UpdateFlags,
241        update_limit: i32,
242    ) -> bool {
243        if update_limit <= 0 {
244            return false;
245        }
246
247        if !self.is_in_valid_bounds(&pos) {
248            return false;
249        }
250
251        let Some(chunk) = self.get_chunk_at(&pos) else {
252            return false;
253        };
254
255        let Some(old_state) = chunk.set_block_state(pos, block_state, flags) else {
256            return false;
257        };
258
259        // Record the block change for broadcasting to clients
260        log::debug!("Block changed at {pos:?}: {old_state:?} -> {block_state:?}");
261        self.chunk_map.block_changed(&pos);
262
263        // Neighbor updates (when UPDATE_NEIGHBORS is set)
264        if flags.contains(UpdateFlags::UPDATE_NEIGHBORS) {
265            self.update_neighbors_at(&pos, old_state.get_block());
266            // TODO: if block has analog output signal, update comparator neighbors
267            // via updateNeighbourForOutputSignal
268        }
269
270        // Shape updates (unless UPDATE_KNOWN_SHAPE is set)
271        if !flags.contains(UpdateFlags::UPDATE_KNOWN_SHAPE) && update_limit > 0 {
272            // Clear UPDATE_NEIGHBORS and UPDATE_SUPPRESS_DROPS for propagation
273            let neighbor_flags =
274                flags & !(UpdateFlags::UPDATE_NEIGHBORS | UpdateFlags::UPDATE_SUPPRESS_DROPS);
275
276            // Notify all 6 neighbors about our shape change
277            for direction in Direction::UPDATE_SHAPE_ORDER {
278                let (dx, dy, dz) = direction.offset();
279                let neighbor_pos = pos.offset(dx, dy, dz);
280
281                // Tell the neighbor that we (at pos) changed
282                self.neighbor_shape_changed(
283                    direction.opposite(), // Direction from us to neighbor
284                    neighbor_pos,         // Neighbor's position
285                    pos,                  // Our position (the one that changed)
286                    block_state,          // Our new state
287                    neighbor_flags,
288                    update_limit - 1,
289                );
290            }
291        }
292
293        true
294    }

296    /// Order in which neighbors are updated (matches vanilla's `NeighborUpdater.UPDATE_ORDER`).
297    const NEIGHBOR_UPDATE_ORDER: [Direction; 6] = [
298        Direction::West,
299        Direction::East,
300        Direction::Down,
301        Direction::Up,
302        Direction::North,
303        Direction::South,
304    ];

306    /// Updates all neighbors of the given position about a block change.
307    ///
308    /// This is the Rust equivalent of vanilla's `Level.updateNeighborsAt()`.
309    fn update_neighbors_at(&self, pos: &BlockPos, source_block: BlockRef) {
310        for direction in Self::NEIGHBOR_UPDATE_ORDER {
311            let (dx, dy, dz) = direction.offset();
312            let neighbor_pos = pos.offset(dx, dy, dz);
313            self.neighbor_changed(neighbor_pos, source_block, false);
314        }
315    }

317    /// Called when a neighbor's shape changes, to update this block's state.
318    ///
319    /// This is the Rust equivalent of vanilla's `NeighborUpdater.executeShapeUpdate()`.
320    fn neighbor_shape_changed(
321        &self,
322        direction: Direction,
323        pos: BlockPos,
324        neighbor_pos: BlockPos,
325        neighbor_state: BlockStateId,
326        flags: UpdateFlags,
327        update_limit: i32,
328    ) {
329        if !self.is_in_valid_bounds(&pos) {
330            return;
331        }
332
333        let current_state = self.get_block_state(&pos);
334
335        // TODO: Skip redstone wire if UPDATE_SKIP_SHAPE_UPDATE_ON_WIRE is set
336        // if flags.contains(UpdateFlags::UPDATE_SKIP_SHAPE_UPDATE_ON_WIRE)
337        //     && current_state.is_redstone_wire() { return; }
338
339        let block_behaviors = BLOCK_BEHAVIORS.get().expect("Behaviors not initialized");
340        let behavior = block_behaviors.get_behavior(current_state.get_block());
341        let new_state = behavior.update_shape(
342            current_state,
343            self,
344            pos,
345            direction,
346            neighbor_pos,
347            neighbor_state,
348        );
349
350        if new_state != current_state {
351            log::debug!(
352                "Shape update at {pos:?}: {current_state:?} -> {new_state:?} (neighbor {neighbor_pos:?} changed)"
353            );
354            // Use set_block_with_limit to prevent infinite recursion
355            self.set_block_with_limit(pos, new_state, flags, update_limit);
356        }
357    }

359    /// Notifies a block that one of its neighbors changed.
360    ///
361    /// This is the Rust equivalent of vanilla's `Level.neighborChanged()`.
362    fn neighbor_changed(&self, pos: BlockPos, source_block: BlockRef, moved_by_piston: bool) {
363        if !self.is_in_valid_bounds(&pos) {
364            return;
365        }
366
367        let state = self.get_block_state(&pos);
368        let block_behaviors = BLOCK_BEHAVIORS.get().expect("Behaviors not initialized");
369        let behavior = block_behaviors.get_behavior(state.get_block());
370        behavior.handle_neighbor_changed(state, self, pos, source_block, moved_by_piston);
371    }

381    /// Ticks the world.
382    pub fn tick_b(&self, tick_count: u64) {
383        self.chunk_map.tick_b(tick_count);
384
385        // Tick players
386        let start = Instant::now();
387        self.players.iter_players(|_uuid, player| {
388            player.tick();
389
390            true
391        });
392        let player_tick_elapsed = start.elapsed();
393        if player_tick_elapsed >= Duration::from_millis(100) {
394            log::warn!("Player tick slow: {player_tick_elapsed:?}");
395        }
396    }

398    /// Broadcasts a signed chat message to all players in the world.
399    ///
400    /// # Panics
401    /// Panics if `message_signature` is `None` after checking `is_some()` (should never happen).
402    pub fn broadcast_chat(
403        &self,
404        mut packet: CPlayerChat,
405        _sender: Arc<Player>,
406        sender_last_seen: LastSeen,
407        message_signature: Option<[u8; 256]>,
408    ) {
409        log::debug!(
410            "broadcast_chat: sender_last_seen has {} signatures, message_signature present: {}",
411            sender_last_seen.len(),
412            message_signature.is_some()
413        );
414
415        self.players.iter_players(|_, recipient| {
416            let messages_received = recipient.get_and_increment_messages_received();
417            packet.global_index = messages_received;
418
419            log::debug!(
420                "Broadcasting to player {} (UUID: {}), global_index={}",
421                recipient.gameprofile.name,
422                recipient.gameprofile.id,
423                messages_received
424            );
425
426            // IMPORTANT: Index previous messages BEFORE updating the cache
427            // This matches vanilla's order: pack() then push()
428            let previous_messages = {
429                let recipient_cache = recipient.signature_cache.lock();
430                recipient_cache.index_previous_messages(&sender_last_seen)
431            };
432
433            log::debug!(
434                "  Indexed {} previous messages for recipient",
435                previous_messages.len()
436            );
437
438            packet.previous_messages.clone_from(&previous_messages);
439
440            // Send the packet
441            recipient.connection.send_packet(packet.clone());
442
443            // AFTER sending, update the recipient's cache using vanilla's push algorithm
444            // This adds all lastSeen signatures + current signature to the cache
445            if let Some(signature) = message_signature {
446                recipient
447                    .signature_cache
448                    .lock()
449                    .push(&sender_last_seen, Some(&signature));
450
451                log::debug!("  Added signature to recipient's cache and pending list");
452
453                // Add to pending messages for acknowledgment tracking
454                recipient
455                    .message_validator
456                    .lock()
457                    .add_pending(Some(Box::new(signature) as Box<[u8]>));
458            } else {
459                // Even unsigned messages update the pending tracker
460                recipient.message_validator.lock().add_pending(None);
461                log::debug!("  Added unsigned message to pending list");
462            }
463
464            true
465        });
466    }

494    /// Broadcasts an encoded packet to all players tracking the given chunk.
495    pub fn broadcast_to_nearby(
496        &self,
497        chunk: ChunkPos,
498        packet: EncodedPacket,
499        exclude: Option<i32>,
500    ) {
501        let tracking_players = self.player_area_map.get_tracking_players(chunk);
502        for entity_id in tracking_players {
503            if Some(entity_id) == exclude {
504                continue;
505            }
506            if let Some(player) = self.players.get_by_entity_id(entity_id) {
507                player.connection.send_encoded_packet(packet.clone());
508            }
509        }
510    }

520    /// Broadcasts block destruction progress to nearby players.
521    ///
522    /// Note: The packet is NOT sent to the player doing the breaking (matching vanilla).
523    /// The breaking player sees progress through client-side prediction.
524    ///
525    /// # Arguments
526    /// * `entity_id` - The entity ID of the player breaking the block
527    /// * `pos` - The position of the block being broken
528    /// * `progress` - The destruction progress (0-9), or -1 to clear
529    #[allow(clippy::cast_sign_loss)]
530    pub fn broadcast_block_destruction(&self, entity_id: i32, pos: BlockPos, progress: i32) {
531        let chunk = ChunkPos::new(
532            SectionPos::block_to_section_coord(pos.x()),
533            SectionPos::block_to_section_coord(pos.z()),
534        );
535        let packet = CBlockDestruction {
536            id: entity_id,
537            pos,
538            progress: progress.clamp(-1, 9) as u8,
539        };
540        let Ok(encoded) =
541            EncodedPacket::from_bare(packet, STEEL_CONFIG.compression, ConnectionProtocol::Play)
542        else {
543            log::warn!("Failed to encode block destruction packet");
544            return;
545        };
546        self.broadcast_to_nearby(chunk, encoded, Some(entity_id));
547    }
