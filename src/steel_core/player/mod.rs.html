212    fn verify_chat_signature(
213        &self,
214        packet: &SChat,
215    ) -> Result<(message_chain::SignedMessageLink, LastSeen), String> {
216        const MESSAGE_EXPIRES_AFTER: Duration = Duration::from_secs(5 * 60);
217
218        let session = self.chat_session.lock().clone().ok_or("No chat session")?;
219        let signature = packet.signature.as_ref().ok_or("No signature present")?;
220
221        if session
222            .profile_public_key
223            .data()
224            .has_expired_with_grace(profile_key::EXPIRY_GRACE_PERIOD)
225        {
226            return Err("Profile key has expired".to_string());
227        }
228
229        let mut chain_guard = self.message_chain.lock();
230        let chain = chain_guard.as_mut().ok_or("No message chain")?;
231
232        if chain.is_broken() {
233            return Err("Message chain is broken".to_string());
234        }
235
236        let timestamp =
237            UNIX_EPOCH + Duration::from_millis(packet.timestamp.try_into().unwrap_or(0));
238
239        let now = SystemTime::now();
240        let message_age = now
241            .duration_since(timestamp)
242            .unwrap_or(Duration::from_secs(0));
243
244        if message_age > MESSAGE_EXPIRES_AFTER {
245            return Err(format!(
246                "Message expired (age: {}s, max: 300s)",
247                message_age.as_secs()
248            ));
249        }
250
251        let last_seen_signatures = self
252            .message_validator
253            .lock()
254            .apply_update(packet.acknowledged, packet.offset, packet.checksum)
255            .map_err(|e| {
256                log::error!("Message acknowledgment validation failed: {e}");
257                e
258            })?;
259
260        let last_seen = LastSeen::new(last_seen_signatures);
261
262        let body = message_chain::SignedMessageBody::new(
263            packet.message.clone(),
264            timestamp,
265            packet.salt,
266            last_seen,
267        );
268
269        let link = chain
270            .validate_and_advance(&body)
271            .map_err(|e| format!("Chain validation failed: {e}"))?;
272
273        let updater = message_chain::MessageSignatureUpdater::new(&link, &body);
274        let validator = session.profile_public_key.create_signature_validator();
275
276        let is_valid =
277            steel_crypto::signature::SignatureValidator::validate(&validator, &updater, signature)
278                .map_err(|e| format!("Signature validation error: {e}"))?;
279
280        if is_valid {
281            Ok((link, body.last_seen.clone()))
282        } else {
283            Err("Invalid signature".to_string())
284        }
285    }

287    /// Handles a chat message from the player.
288    #[allow(clippy::too_many_lines)]
289    pub fn handle_chat(&self, packet: SChat, player: Arc<Player>) {
290        let chat_message = packet.message.clone();
291
292        let verification_result = if let Some(_signature) = &packet.signature {
293            match self.verify_chat_signature(&packet) {
294                Ok((link, last_seen)) => {
295                    // Don't add to cache here - will be added during broadcast
296                    // to avoid cache state mismatch with client
297                    Some(Ok((link, last_seen)))
298                }
299                Err(err) => {
300                    log::warn!(
301                        "Player {} sent message with invalid signature: {err}",
302                        self.gameprofile.name
303                    );
304                    Some(Err(err))
305                }
306            }
307        } else {
308            None
309        };
310
311        if STEEL_CONFIG.enforce_secure_chat {
312            match &verification_result {
313                Some(Ok(_)) => {}
314                Some(Err(err)) => {
315                    self.connection.disconnect(
316                        TextComponent::new().text(format!("Chat message validation failed: {err}")),
317                    );
318                    return;
319                }
320                None => {
321                    self.connection.disconnect(TextComponent::new().text(
322                        "Secure chat is enforced on this server, but your message was not signed",
323                    ));
324                    return;
325                }
326            }
327        }
328
329        let signature = if matches!(verification_result, Some(Ok(_))) {
330            packet.signature.map(|sig| Box::new(sig) as Box<[u8]>)
331        } else {
332            None
333        };
334
335        let sender_index = player
336            .messages_sent
337            .fetch_add(1, std::sync::atomic::Ordering::SeqCst);
338
339        let chat_packet = CPlayerChat::new(
340            0,
341            player.gameprofile.id,
342            sender_index,
343            signature.clone(),
344            chat_message.clone(),
345            packet.timestamp,
346            packet.salt,
347            Box::new([]),
348            Some(TextComponent::new().text(chat_message.clone())),
349            FilterType::PassThrough,
350            steel_protocol::packets::game::ChatTypeBound {
351                //TODO: Use the registry to derive this instead of hardcoding it
352                registry_id: 0,
353                sender_name: TextComponent::new().text(player.gameprofile.name.clone()),
354                target_name: None,
355            },
356        );
357
358        if let Some(ref sig_box) = signature {
359            if sig_box.len() == 256 {
360                let mut sig_array = [0u8; 256];
361                sig_array.copy_from_slice(&sig_box[..]);
362
363                let last_seen = if let Some(Ok((_, ref last_seen))) = verification_result {
364                    last_seen.clone()
365                } else {
366                    LastSeen::default()
367                };
368
369                log::info!("<{}> {}", player.gameprofile.name, chat_message);
370                self.world.broadcast_chat(
371                    chat_packet,
372                    Arc::clone(&player),
373                    last_seen,
374                    Some(sig_array),
375                );
376            } else {
377                self.world.broadcast_unsigned_chat(
378                    chat_packet,
379                    &player.gameprofile.name,
380                    &chat_message,
381                );
382            }
383        } else {
384            self.world.broadcast_unsigned_chat(
385                chat_packet,
386                &player.gameprofile.name,
387                &chat_message,
388            );
389        }
390    }

432    /// Updates the player's chat session and initializes the message chain.
433    ///
434    /// This should be called when receiving a `ChatSessionUpdate` packet from the client.
435    pub fn set_chat_session(&self, session: RemoteChatSession) {
436        // Initialize the message chain for this session
437        let chain = SignedMessageChain::new(self.gameprofile.id, session.session_id);
438
439        // Convert session to data for broadcasting
440        let session_data = session.as_data();
441        let protocol_data = match session_data.to_protocol_data() {
442            Ok(data) => data,
443            Err(err) => {
444                log::error!(
445                    "Failed to convert chat session to protocol data for {}: {:?}",
446                    self.gameprofile.name,
447                    err
448                );
449                *self.chat_session.lock() = Some(session);
450                *self.message_chain.lock() = Some(chain);
451                return;
452            }
453        };
454
455        *self.chat_session.lock() = Some(session);
456        *self.message_chain.lock() = Some(chain);
457
458        log::info!(
459            "Player {} initialized signed chat session",
460            self.gameprofile.name
461        );
462
463        // Broadcast the chat session to all players so they can verify this player's signatures
464        let update_packet = steel_protocol::packets::game::CPlayerInfoUpdate::update_chat_session(
465            self.gameprofile.id,
466            protocol_data,
467        );
468
469        self.world.players.iter_sync(|_, player| {
470            player.connection.send_packet(update_packet.clone());
471            true
472        });
473    }

485    /// Handles a chat session update packet from the client.
486    ///
487    /// This validates the player's profile key and initializes signed chat if valid.
488    pub fn handle_chat_session_update(&self, packet: SChatSessionUpdate) {
489        log::info!("Player {} sent chat session update", self.gameprofile.name);
490
491        // Convert the packet data to profile key data
492        let expires_at = UNIX_EPOCH + Duration::from_millis(packet.expires_at as u64);
493
494        // Decode the public key
495        let public_key = match steel_crypto::public_key_from_bytes(&packet.public_key) {
496            Ok(key) => key,
497            Err(err) => {
498                log::warn!(
499                    "Player {} sent invalid public key: {err}",
500                    self.gameprofile.name
501                );
502                // Phase 4: Kick if enforcement is enabled
503                if STEEL_CONFIG.enforce_secure_chat {
504                    log::error!(
505                        "Player {} kicked for invalid public key",
506                        self.gameprofile.name
507                    );
508                    self.connection
509                        .disconnect(TextComponent::new().text("Invalid profile public key"));
510                }
511                return;
512            }
513        };
514
515        let profile_key_data =
516            profile_key::ProfilePublicKeyData::new(expires_at, public_key, packet.key_signature);
517
518        let validator = Box::new(steel_crypto::signature::NoValidation)
519            as Box<dyn steel_crypto::SignatureValidator>;
520
521        let session_data = profile_key::RemoteChatSessionData {
522            session_id: packet.session_id,
523            profile_public_key: profile_key_data,
524        };
525
526        match session_data.validate(self.gameprofile.id, &*validator) {
527            Ok(session) => {
528                self.set_chat_session(session);
529            }
530            Err(err) => {
531                log::warn!(
532                    "Player {} sent invalid chat session: {err}",
533                    self.gameprofile.name
534                );
535                if STEEL_CONFIG.enforce_secure_chat {
536                    self.connection.disconnect(
537                        TextComponent::new().text(format!("Chat session validation failed: {err}")),
538                    );
539                }
540            }
541        }
542    }

591    /// Handles a container click packet (slot interaction).
592    pub fn handle_container_click(&self, packet: SContainerClick) {
593        let mut menu = self.inventory_menu.lock();
594
595        // Check container ID matches
596        if i32::from(menu.behavior().container_id) != packet.container_id {
597            return;
598        }
599
600        // Handle spectator mode - just resync the inventory
601        if self.game_mode.load() == GameType::Spectator {
602            menu.behavior_mut()
603                .send_all_data_to_remote(&self.connection);
604            return;
605        }
606
607        // Validate slot index
608        if !menu.behavior().is_valid_slot_index(packet.slot_num) {
609            log::debug!(
610                "Player {} clicked invalid slot index: {}, available: {}",
611                self.gameprofile.name,
612                packet.slot_num,
613                menu.behavior().slot_count()
614            );
615            return;
616        }
617
618        // Check if we need a full resync (state ID mismatch)
619        let full_resync_needed = packet.state_id as u32 != menu.behavior().get_state_id();
620
621        // Suppress remote updates during click handling
622        menu.behavior_mut().suppress_remote_updates();
623
624        // Handle the click using the Menu trait method
625        let has_infinite_materials = self.game_mode.load() == GameType::Creative;
626        menu.clicked(
627            packet.slot_num,
628            packet.button_num,
629            packet.click_type,
630            has_infinite_materials,
631            self,
632        );
633
634        // Update remote slots from the client's perception
635        for (slot, hash) in packet.changed_slots {
636            menu.behavior_mut().set_remote_slot(slot as usize, hash);
637        }
638
639        // Update remote carried from the client's perception
640        menu.behavior_mut().set_remote_carried(packet.carried_item);
641
642        // Resume remote updates
643        menu.behavior_mut().resume_remote_updates();
644
645        // Broadcast changes or full state depending on whether we had a state mismatch
646        if full_resync_needed {
647            menu.behavior_mut().broadcast_full_state(&self.connection);
648        } else {
649            menu.behavior_mut().broadcast_changes(&self.connection);
650        }
651    }

685    /// Handles a creative mode slot set packet.
686    pub fn handle_set_creative_mode_slot(&self, packet: SSetCreativeModeSlot) {
687        // Only allow in creative mode
688        if self.game_mode.load() != GameType::Creative {
689            return;
690        }
691
692        let drop = packet.slot_num < 0;
693        let item_stack = packet.item_stack;
694
695        // Validate slot range (1-45 for inventory menu)
696        let valid_slot = packet.slot_num >= 1 && packet.slot_num <= 45;
697        let valid_data = item_stack.is_empty() || item_stack.count <= item_stack.max_stack_size();
698
699        if valid_slot && valid_data {
700            let mut menu = self.inventory_menu.lock();
701            let slot_index = packet.slot_num as usize;
702
703            if let Some(slot) = menu.behavior().get_slot(slot_index) {
704                slot.set_item(item_stack.clone());
705            }
706
707            menu.behavior_mut()
708                .set_remote_slot_known(slot_index, &item_stack);
709            menu.behavior_mut().broadcast_changes(&self.connection);
710        } else if drop && valid_data {
711            // TODO: Implement drop spam throttling
712            // For now, just drop the item
713            if !item_stack.is_empty() {
714                // TODO: Actually drop the item into the world
715                log::debug!(
716                    "Player {} would drop {:?} in creative mode",
717                    self.gameprofile.name,
718                    item_stack
719                );
720            }
721        }
722    }

773    /// Tries to add an item to the player's inventory, dropping it if it doesn't fit.
774    ///
775    /// Based on Java's `Inventory.placeItemBackInInventory`.
776    pub fn add_item_or_drop(&self, mut item: steel_registry::item_stack::ItemStack) {
777        if item.is_empty() {
778            return;
779        }
780
781        // Try to add to inventory
782        let added = self.inventory.lock().add(&mut item);
783        if !added || !item.is_empty() {
784            // Couldn't fit everything, drop the rest
785            self.drop_item(item, false);
786        }
787    }
