247    fn verify_chat_signature(
248        &self,
249        packet: &SChat,
250    ) -> Result<(message_chain::SignedMessageLink, LastSeen), String> {
251        const MESSAGE_EXPIRES_AFTER: Duration = Duration::from_secs(5 * 60);
252
253        let session = self.chat_session.lock().clone().ok_or("No chat session")?;
254        let signature = packet.signature.as_ref().ok_or("No signature present")?;
255
256        if session
257            .profile_public_key
258            .data()
259            .has_expired_with_grace(profile_key::EXPIRY_GRACE_PERIOD)
260        {
261            return Err("Profile key has expired".to_string());
262        }
263
264        let mut chain_guard = self.message_chain.lock();
265        let chain = chain_guard.as_mut().ok_or("No message chain")?;
266
267        if chain.is_broken() {
268            return Err("Message chain is broken".to_string());
269        }
270
271        let timestamp =
272            UNIX_EPOCH + Duration::from_millis(packet.timestamp.try_into().unwrap_or(0));
273
274        let now = SystemTime::now();
275        let message_age = now
276            .duration_since(timestamp)
277            .unwrap_or(Duration::from_secs(0));
278
279        if message_age > MESSAGE_EXPIRES_AFTER {
280            return Err(format!(
281                "Message expired (age: {}s, max: 300s)",
282                message_age.as_secs()
283            ));
284        }
285
286        let last_seen_signatures = self
287            .message_validator
288            .lock()
289            .apply_update(packet.acknowledged, packet.offset, packet.checksum)
290            .map_err(|e| {
291                log::error!("Message acknowledgment validation failed: {e}");
292                e
293            })?;
294
295        let last_seen = LastSeen::new(last_seen_signatures);
296
297        let body = message_chain::SignedMessageBody::new(
298            packet.message.clone(),
299            timestamp,
300            packet.salt,
301            last_seen,
302        );
303
304        let link = chain
305            .validate_and_advance(&body)
306            .map_err(|e| format!("Chain validation failed: {e}"))?;
307
308        let updater = message_chain::MessageSignatureUpdater::new(&link, &body);
309        let validator = session.profile_public_key.create_signature_validator();
310
311        let is_valid = SignatureValidator::validate(&validator, &updater, signature)
312            .map_err(|e| format!("Signature validation error: {e}"))?;
313
314        if is_valid {
315            Ok((link, body.last_seen.clone()))
316        } else {
317            Err("Invalid signature".to_string())
318        }
319    }

321    /// Handles a chat message from the player.
322    #[allow(clippy::too_many_lines)]
323    pub fn handle_chat(&self, packet: SChat, player: Arc<Player>) {
324        let chat_message = packet.message.clone();
325
326        let verification_result = if let Some(_signature) = &packet.signature {
327            match self.verify_chat_signature(&packet) {
328                Ok((link, last_seen)) => {
329                    // Don't add to cache here - will be added during broadcast
330                    // to avoid cache state mismatch with client
331                    Some(Ok((link, last_seen)))
332                }
333                Err(err) => {
334                    log::warn!(
335                        "Player {} sent message with invalid signature: {err}",
336                        self.gameprofile.name
337                    );
338                    Some(Err(err))
339                }
340            }
341        } else {
342            None
343        };
344
345        if STEEL_CONFIG.enforce_secure_chat {
346            match &verification_result {
347                Some(Ok(_)) => {}
348                Some(Err(err)) => {
349                    self.connection.disconnect(
350                        TextComponent::new().text(format!("Chat message validation failed: {err}")),
351                    );
352                    return;
353                }
354                None => {
355                    self.connection.disconnect(TextComponent::new().text(
356                        "Secure chat is enforced on this server, but your message was not signed",
357                    ));
358                    return;
359                }
360            }
361        }
362
363        let signature = if matches!(verification_result, Some(Ok(_))) {
364            packet.signature.map(|sig| Box::new(sig) as Box<[u8]>)
365        } else {
366            None
367        };
368
369        let sender_index = player.messages_sent.fetch_add(1, Ordering::SeqCst);
370
371        let chat_packet = CPlayerChat::new(
372            0,
373            player.gameprofile.id,
374            sender_index,
375            signature.clone(),
376            chat_message.clone(),
377            packet.timestamp,
378            packet.salt,
379            Box::new([]),
380            Some(TextComponent::new().text(chat_message.clone())),
381            FilterType::PassThrough,
382            ChatTypeBound {
383                //TODO: Use the registry to derive this instead of hardcoding it
384                registry_id: 0,
385                sender_name: TextComponent::new().text(player.gameprofile.name.clone()),
386                target_name: None,
387            },
388        );
389
390        if let Some(ref sig_box) = signature {
391            if sig_box.len() == 256 {
392                let mut sig_array = [0u8; 256];
393                sig_array.copy_from_slice(&sig_box[..]);
394
395                let last_seen = if let Some(Ok((_, ref last_seen))) = verification_result {
396                    last_seen.clone()
397                } else {
398                    LastSeen::default()
399                };
400
401                log::info!("<{}> {}", player.gameprofile.name, chat_message);
402                self.world.broadcast_chat(
403                    chat_packet,
404                    Arc::clone(&player),
405                    last_seen,
406                    Some(sig_array),
407                );
408            } else {
409                self.world.broadcast_unsigned_chat(
410                    chat_packet,
411                    &player.gameprofile.name,
412                    &chat_message,
413                );
414            }
415        } else {
416            self.world.broadcast_unsigned_chat(
417                chat_packet,
418                &player.gameprofile.name,
419                &chat_message,
420            );
421        }
422    }

442    /// Handles a move player packet.
443    pub fn handle_move_player(&self, packet: SMovePlayer) {
444        if Self::is_invalid_position(
445            packet.get_x(0.0),
446            packet.get_y(0.0),
447            packet.get_z(0.0),
448            packet.get_x_rot(0.0),
449            packet.get_y_rot(0.0),
450        ) {
451            self.connection
452                .disconnect(translations::MULTIPLAYER_DISCONNECT_INVALID_PLAYER_MOVEMENT.msg());
453            return;
454        }
455
456        if self.update_awaiting_teleport() || !self.client_loaded.load(Ordering::Relaxed) {
457            return;
458        }
459
460        let prev_pos = *self.prev_position.lock();
461        let prev_rot = self.prev_rotation.load();
462
463        // Update current state
464        if packet.has_pos {
465            *self.position.lock() = packet.position;
466        }
467        if packet.has_rot {
468            self.rotation.store((packet.y_rot, packet.x_rot));
469        }
470
471        // Broadcast movement to other players
472        let pos = if packet.has_pos {
473            packet.position
474        } else {
475            prev_pos
476        };
477        let (yaw, pitch) = if packet.has_rot {
478            (packet.y_rot, packet.x_rot)
479        } else {
480            prev_rot
481        };
482
483        if packet.has_pos || packet.has_rot {
484            let new_chunk = ChunkPos::new((pos.x as i32) >> 4, (pos.z as i32) >> 4);
485
486            // Note: player_area_map is updated in chunk_map.update_player_status
487            // which is called every tick and computes view diffs efficiently
488
489            if packet.has_pos {
490                let move_packet = CMoveEntityPosRot {
491                    entity_id: self.entity_id,
492                    dx: calc_delta(pos.x, prev_pos.x),
493                    dy: calc_delta(pos.y, prev_pos.y),
494                    dz: calc_delta(pos.z, prev_pos.z),
495                    y_rot: to_angle_byte(yaw),
496                    x_rot: to_angle_byte(pitch),
497                    on_ground: packet.on_ground,
498                };
499                self.world
500                    .broadcast_to_nearby(new_chunk, move_packet, Some(self.gameprofile.id));
501            } else {
502                let rot_packet = CMoveEntityRot {
503                    entity_id: self.entity_id,
504                    y_rot: to_angle_byte(yaw),
505                    x_rot: to_angle_byte(pitch),
506                    on_ground: packet.on_ground,
507                };
508                self.world
509                    .broadcast_to_nearby(new_chunk, rot_packet, Some(self.gameprofile.id));
510            }
511
512            if packet.has_rot {
513                let head_packet = CRotateHead {
514                    entity_id: self.entity_id,
515                    head_y_rot: to_angle_byte(yaw),
516                };
517                self.world
518                    .broadcast_to_nearby(new_chunk, head_packet, Some(self.gameprofile.id));
519            }
520
521            *self.prev_position.lock() = pos;
522            self.prev_rotation.store((yaw, pitch));
523        }
524    }

526    /// Updates the player's chat session and initializes the message chain.
527    ///
528    /// This should be called when receiving a `ChatSessionUpdate` packet from the client.
529    pub fn set_chat_session(&self, session: RemoteChatSession) {
530        // Initialize the message chain for this session
531        let chain = SignedMessageChain::new(self.gameprofile.id, session.session_id);
532
533        // Convert session to data for broadcasting
534        let session_data = session.as_data();
535        let protocol_data = match session_data.to_protocol_data() {
536            Ok(data) => data,
537            Err(err) => {
538                log::error!(
539                    "Failed to convert chat session to protocol data for {}: {:?}",
540                    self.gameprofile.name,
541                    err
542                );
543                *self.chat_session.lock() = Some(session);
544                *self.message_chain.lock() = Some(chain);
545                return;
546            }
547        };
548
549        *self.chat_session.lock() = Some(session);
550        *self.message_chain.lock() = Some(chain);
551
552        log::info!(
553            "Player {} initialized signed chat session",
554            self.gameprofile.name
555        );
556
557        // Broadcast the chat session to all players so they can verify this player's signatures
558        let update_packet =
559            CPlayerInfoUpdate::update_chat_session(self.gameprofile.id, protocol_data);
560
561        self.world.players.iter_sync(|_, player| {
562            player.connection.send_packet(update_packet.clone());
563            true
564        });
565    }

577    /// Handles a chat session update packet from the client.
578    ///
579    /// This validates the player's profile key and initializes signed chat if valid.
580    pub fn handle_chat_session_update(&self, packet: SChatSessionUpdate) {
581        log::info!("Player {} sent chat session update", self.gameprofile.name);
582
583        // Convert the packet data to profile key data
584        let expires_at = UNIX_EPOCH + Duration::from_millis(packet.expires_at as u64);
585
586        // Decode the public key
587        let public_key = match public_key_from_bytes(&packet.public_key) {
588            Ok(key) => key,
589            Err(err) => {
590                log::warn!(
591                    "Player {} sent invalid public key: {err}",
592                    self.gameprofile.name
593                );
594                // Phase 4: Kick if enforcement is enabled
595                if STEEL_CONFIG.enforce_secure_chat {
596                    log::error!(
597                        "Player {} kicked for invalid public key",
598                        self.gameprofile.name
599                    );
600                    self.connection
601                        .disconnect(TextComponent::new().text("Invalid profile public key"));
602                }
603                return;
604            }
605        };
606
607        let profile_key_data =
608            profile_key::ProfilePublicKeyData::new(expires_at, public_key, packet.key_signature);
609
610        let validator = Box::new(NoValidation) as Box<dyn SignatureValidator>;
611
612        let session_data = profile_key::RemoteChatSessionData {
613            session_id: packet.session_id,
614            profile_public_key: profile_key_data,
615        };
616
617        match session_data.validate(self.gameprofile.id, &*validator) {
618            Ok(session) => {
619                self.set_chat_session(session);
620            }
621            Err(err) => {
622                log::warn!(
623                    "Player {} sent invalid chat session: {err}",
624                    self.gameprofile.name
625                );
626                if STEEL_CONFIG.enforce_secure_chat {
627                    self.connection.disconnect(
628                        TextComponent::new().text(format!("Chat session validation failed: {err}")),
629                    );
630                }
631            }
632        }
633    }

680    /// Handles a container click packet (slot interaction).
681    pub fn handle_container_click(&self, packet: SContainerClick) {
682        let mut menu = self.inventory_menu.lock();
683
684        // Check container ID matches
685        if i32::from(menu.behavior().container_id) != packet.container_id {
686            return;
687        }
688
689        // Handle spectator mode - just resync the inventory
690        if self.game_mode.load() == GameType::Spectator {
691            menu.behavior_mut()
692                .send_all_data_to_remote(&self.connection);
693            return;
694        }
695
696        // Validate slot index
697        if !menu.behavior().is_valid_slot_index(packet.slot_num) {
698            log::debug!(
699                "Player {} clicked invalid slot index: {}, available: {}",
700                self.gameprofile.name,
701                packet.slot_num,
702                menu.behavior().slot_count()
703            );
704            return;
705        }
706
707        // Check if we need a full resync (state ID mismatch)
708        let full_resync_needed = packet.state_id as u32 != menu.behavior().get_state_id();
709
710        // Suppress remote updates during click handling
711        menu.behavior_mut().suppress_remote_updates();
712
713        // Handle the click using the Menu trait method
714        let has_infinite_materials = self.game_mode.load() == GameType::Creative;
715        menu.clicked(
716            packet.slot_num,
717            packet.button_num,
718            packet.click_type,
719            has_infinite_materials,
720            self,
721        );
722
723        // Update remote slots from the client's perception
724        for (slot, hash) in packet.changed_slots {
725            menu.behavior_mut().set_remote_slot(slot as usize, hash);
726        }
727
728        // Update remote carried from the client's perception
729        menu.behavior_mut().set_remote_carried(packet.carried_item);
730
731        // Resume remote updates
732        menu.behavior_mut().resume_remote_updates();
733
734        // Broadcast changes or full state depending on whether we had a state mismatch
735        if full_resync_needed {
736            menu.behavior_mut().broadcast_full_state(&self.connection);
737        } else {
738            menu.behavior_mut().broadcast_changes(&self.connection);
739        }
740    }

825    /// Returns true if player is within block interaction range.
826    /// Base range is ~4.5 blocks, plus 1.0 tolerance.
827    #[must_use]
828    pub fn is_within_block_interaction_range(&self, pos: &BlockPos) -> bool {
829        let player_pos = *self.position.lock();
830        let block_center_x = f64::from(pos.x()) + 0.5;
831        let block_center_y = f64::from(pos.y()) + 0.5;
832        let block_center_z = f64::from(pos.z()) + 0.5;
833
834        // Base range is ~4.5 blocks, plus 1.0 tolerance
835        let max_range = 4.5 + 1.0;
836        let dx = player_pos.x - block_center_x;
837        let dy = player_pos.y - block_center_y;
838        let dz = player_pos.z - block_center_z;
839        (dx * dx + dy * dy + dz * dz).sqrt() <= max_range
840    }

899    /// Handles the use of an item on a block.
900    ///
901    /// Implements the logic from Java's `ServerGamePacketListenerImpl.handleUseItemOn()`.
902    pub fn handle_use_item_on(&self, packet: SUseItemOn) {
903        // 1. Check client loaded
904        if !self.client_loaded.load(Ordering::Relaxed) {
905            return;
906        }
907
908        // 2. Ack block changes
909        self.ack_block_changes_up_to(packet.sequence);
910
911        let pos = &packet.block_hit.block_pos;
912        let direction = packet.block_hit.direction;
913
914        // 3. Validate interaction range
915        if !self.is_within_block_interaction_range(pos) {
916            self.send_block_updates(pos, direction);
917            return;
918        }
919
920        // 4. Validate hit location precision (must be within 1.0000001 of block center)
921        let center_x = f64::from(pos.x()) + 0.5;
922        let center_y = f64::from(pos.y()) + 0.5;
923        let center_z = f64::from(pos.z()) + 0.5;
924        let location = &packet.block_hit.location;
925        let limit = 1.000_000_1;
926
927        if (location.x - center_x).abs() >= limit
928            || (location.y - center_y).abs() >= limit
929            || (location.z - center_z).abs() >= limit
930        {
931            log::warn!(
932                "Rejecting UseItemOnPacket from {}: location {:?} too far from block {:?}",
933                self.gameprofile.name,
934                location,
935                pos
936            );
937            self.send_block_updates(pos, direction);
938            return;
939        }
940
941        // 5. Validate Y height
942        if pos.y() >= self.world.max_build_height() {
943            // TODO: Send "build.tooHigh" message to player
944            self.send_block_updates(pos, direction);
945            return;
946        }
947
948        // 6. Check awaiting teleport
949        if self.is_awaiting_teleport() {
950            self.send_block_updates(pos, direction);
951            return;
952        }
953
954        // 7. Check may_interact permission
955        if !self.world.may_interact(self, pos) {
956            self.send_block_updates(pos, direction);
957            return;
958        }
959
960        // 8. Call use_item_on
961        let result = game_mode::use_item_on(self, &self.world, packet.hand, &packet.block_hit);
962
963        // 9. Handle result
964        if let InteractionResult::Success = result {
965            // TODO: Trigger arm swing animation if needed
966            self.swing(packet.hand, true);
967        }
968
969        // 10. Always send block updates to resync client
970        self.send_block_updates(pos, direction);
971
972        // 11. Broadcast inventory changes
973        self.broadcast_inventory_changes();
974    }

1043    /// Tries to add an item to the player's inventory, dropping it if it doesn't fit.
1044    ///
1045    /// Based on Java's `Inventory.placeItemBackInInventory`.
1046    pub fn add_item_or_drop(&self, mut item: ItemStack) {
1047        if item.is_empty() {
1048            return;
1049        }
1050
1051        // Try to add to inventory
1052        let added = self.inventory.lock().add(&mut item);
1053        if !added || !item.is_empty() {
1054            // Couldn't fit everything, drop the rest
1055            self.drop_item(item, false);
1056        }
1057    }

1059    /// Tries to add an item to the player's inventory using an existing lock guard,
1060    /// dropping it if it doesn't fit.
1061    ///
1062    /// Use this variant when you already hold a `ContainerLockGuard` that includes
1063    /// the player's inventory to avoid deadlocks.
1064    pub fn add_item_or_drop_with_guard(&self, guard: &mut ContainerLockGuard, mut item: ItemStack) {
1065        if item.is_empty() {
1066            return;
1067        }
1068
1069        let inv_id = ContainerId::from_arc(&self.inventory);
1070        if let Some(inv) = guard.get_mut(inv_id) {
1071            let added = inv.add(&mut item);
1072            if !added || !item.is_empty() {
1073                self.drop_item(item, false);
1074            }
1075        } else {
1076            // Inventory not in guard - this shouldn't happen but drop the item to be safe
1077            self.drop_item(item, false);
1078        }
1079    }
