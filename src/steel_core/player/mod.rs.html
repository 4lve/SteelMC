128    /// Handles a move player packet.
129    pub fn handle_move_player(&self, packet: SMovePlayer) {
130        if Self::is_invalid_position(
131            packet.get_x(0.0),
132            packet.get_y(0.0),
133            packet.get_z(0.0),
134            packet.get_x_rot(0.0),
135            packet.get_y_rot(0.0),
136        ) {
137            self.connection
138                .disconnect(translations::MULTIPLAYER_DISCONNECT_INVALID_PLAYER_MOVEMENT.msg());
139            return;
140        }
141
142        if !self.update_awaiting_teleport()
143            && self.client_loaded.load(Ordering::Relaxed)
144            && packet.has_pos
145        {
146            *self.position.lock() = packet.position;
