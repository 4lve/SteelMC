1//! Game mode specific logic for player interactions.
2//!
3//! This module implements the logic from Java's `ServerPlayerGameMode`, particularly
4//! the `useItemOn` method that handles block placement and block interactions.
5
6use steel_registry::REGISTRY;
7use steel_registry::items::item::{BlockHitResult, InteractionResult, UseOnContext};
8use steel_utils::types::{GameType, InteractionHand};
9
10use crate::player::Player;
11use crate::world::World;
12
13/// Handles using an item on a block.
14///
15/// This implements the logic from Java's `ServerPlayerGameMode.useItemOn()`.
16///
17/// # Flow
18/// 1. Spectator mode: Only allow opening menus (currently returns Pass)
19/// 2. Check if block interaction should be suppressed (sneaking + holding items)
20/// 3. If not suppressed: Call block's `use_item_on` method
21/// 4. If block returns `TryEmptyHandInteraction` and main hand: Call block's `use_without_item`
22/// 5. If item not empty: Call item behavior's `use_on` for placement
23/// 6. Handle creative mode infinite materials
24pub fn use_item_on(
25    player: &Player,
26    world: &World,
27    hand: InteractionHand,
28    hit_result: &BlockHitResult,
29) -> InteractionResult {
30    let pos = &hit_result.block_pos;
31    let state = world.get_block_state(pos);
32
33    // Spectator mode: can only open menus
34    // TODO: Implement menu providers for blocks like chests
35    if player.game_mode.load() == GameType::Spectator {
36        return InteractionResult::Pass;
37    }
38
39    // Check if block interaction should be suppressed (sneaking + holding items in either hand)
40    let mut inv = player.inventory.lock();
41
42    let have_something = !inv.get_item_in_hand(InteractionHand::MainHand).is_empty()
43        || !inv.get_item_in_hand(InteractionHand::OffHand).is_empty();
44    let suppress_block_use = player.is_secondary_use_active() && have_something;
45
46    // Try block interaction first (if not suppressed)
47    if !suppress_block_use {
48        // Get block behavior and call use_item_on
49        let Some(block) = REGISTRY.blocks.by_state_id(state) else {
50            // Block state not found in registry, skip block interaction
51            return InteractionResult::Pass;
52        };
53        let behavior = REGISTRY.blocks.get_behavior(block);
54        let item_stack = inv.get_item_in_hand(hand);
55
56        let block_result =
57            behavior.use_item_on(item_stack, state, world, *pos, player, hand, hit_result);
58
59        if block_result.consumes_action() {
60            return block_result;
61        }
62
63        // Try empty hand interaction for main hand if block requested it
64        if matches!(block_result, InteractionResult::TryEmptyHandInteraction)
65            && hand == InteractionHand::MainHand
66        {
67            let empty_result = behavior.use_without_item(state, world, *pos, player, hit_result);
68
69            if empty_result.consumes_action() {
70                return empty_result;
71            }
72        }
73    }
74
75    // Try item use (block placement, etc.)
76    let item_stack = inv.get_item_in_hand_mut(hand);
77    if !item_stack.is_empty() {
78        // TODO: Check item cooldowns
79        // if player.getCooldowns().isOnCooldown(item_stack.item) { return Pass }
80
81        let original_count = item_stack.count;
82
83        let mut context = UseOnContext {
84            player,
85            hand,
86            hit_result: hit_result.clone(),
87            world,
88            item_stack,
89        };
90
91        // Get item behavior and call use_on
92        let item_behavior = REGISTRY.items.get_behavior(context.item_stack.item);
93        let result = item_behavior.use_on(&mut context);
94
95        // Restore count for creative mode (infinite materials)
96        if player.has_infinite_materials() && context.item_stack.count < original_count {
97            context.item_stack.count = original_count;
98        }
99
100        return result;
101    }
102
103    InteractionResult::Pass
104}
