25/// A chunk that is ready to be sent to the client.
26///
27/// Similar to Java's `LevelChunk`, this holds a weak reference to the world
28/// (called `level` in Java) for callbacks during block state changes.
29pub struct LevelChunk {
30    /// The sections of the chunk.
31    pub sections: Sections,
32    /// The position of the chunk.
33    pub pos: ChunkPos,
34    /// Whether the chunk has been modified since last save.
35    pub dirty: Arc<AtomicBool>,
36    /// The heightmaps for this chunk (wrapped in `RwLock` for interior mutability).
37    pub heightmaps: Arc<SyncRwLock<ChunkHeightmaps>>,
38    /// The minimum Y coordinate of the world this chunk belongs to.
39    min_y: i32,
40    /// The total height of the world.
41    height: i32,
42    /// Weak reference to the world (called `level` in Java).
43    /// This mirrors Java's `LevelChunk.level` field.
44    level: Weak<World>,
45}
46
47impl LevelChunk {
48    /// Creates a new `LevelChunk` from a `ProtoChunk`.
49    ///
50    /// Transfers final heightmaps from the proto chunk if available.
51    ///
52    /// # Arguments
53    /// * `proto_chunk` - The proto chunk to convert
54    /// * `min_y` - The minimum Y coordinate of the world
55    /// * `height` - The total height of the world
56    /// * `level` - Weak reference to the world (mirrors Java's `LevelChunk.level`)
57    #[must_use]
58    pub fn from_proto(
59        proto_chunk: ProtoChunk,
60        min_y: i32,
61        height: i32,
62        level: Weak<World>,
63    ) -> Self {
64        // Transfer final heightmaps from proto chunk if available
65        let proto_heightmaps = proto_chunk.heightmaps.read();
66        let mut chunk_heightmaps = ChunkHeightmaps::new(min_y, height);
67
68        // Copy final heightmap data if available in proto chunk
69        for &hm_type in HeightmapType::final_types() {
70            if let Some(proto_hm) = proto_heightmaps.get(&hm_type) {
71                chunk_heightmaps
72                    .get_mut(hm_type)
73                    .set_raw_data(&proto_hm.get_raw_data());
74            }
75        }
76        drop(proto_heightmaps);
77
78        Self {
79            sections: proto_chunk.sections,
80            pos: proto_chunk.pos,
81            dirty: proto_chunk.dirty.clone(),
82            heightmaps: Arc::new(SyncRwLock::new(chunk_heightmaps)),
83            min_y,
84            height,
85            level,
86        }
87    }

136    /// Gets the section index for a given Y coordinate.
137    #[must_use]
138    fn get_section_index(&self, y: i32) -> usize {
139        ((y - self.min_y) / 16) as usize
140    }

142    /// Marks the chunk as unsaved.
143    fn mark_unsaved(&self) {
144        self.dirty.store(true, Ordering::Release);

147    /// Sets a block state at the given position.
148    ///
149    /// Returns the old block state, or `None` if nothing changed.
150    ///
151    /// # Arguments
152    /// * `pos` - The absolute block position
153    /// * `state` - The new block state to set
154    /// * `flags` - Update flags controlling behavior
155    #[must_use]
156    pub fn set_block_state(
157        &self,
158        pos: BlockPos,
159        state: BlockStateId,
160        flags: UpdateFlags,
161    ) -> Option<BlockStateId> {
162        let y = pos.0.y;
163
164        if y < self.min_y || y >= self.min_y + self.height {
165            return None;
166        }
167
168        let section_index = self.get_section_index(y);
169
170        if section_index >= self.sections.sections.len() {
171            return None;
172        }
173
174        let section = &self.sections.sections[section_index];
175
176        let was_empty = section.read().states.has_only_air();
177        if was_empty && state.is_air() {
178            return None;
179        }
180
181        let local_x = (pos.0.x & 15) as usize;
182        let local_y = (y & 15) as usize;
183        let local_z = (pos.0.z & 15) as usize;
184
185        let old_state = section.write().states.set(local_x, local_y, local_z, state);
186
187        if old_state == state {
188            return None;
189        }
190
191        let min_y = self.min_y;
192        let sections = &self.sections;
193        self.heightmaps
194            .write()
195            .update(local_x, y, local_z, state, |lx, scan_y, lz| {
196                let scan_section_index = ((scan_y - min_y) / 16) as usize;
197                let scan_local_y = ((scan_y - min_y) % 16) as usize;
198                sections.sections[scan_section_index]
199                    .read()
200                    .states
201                    .get(lx, scan_local_y, lz)
202            });
203
204        let old_block = old_state.get_block();
205        let new_block = state.get_block();
206
207        let current_block = section
208            .read()
209            .states
210            .get(local_x, local_y, local_z)
211            .get_block();
212        if !std::ptr::eq(current_block, new_block) {
213            return None;
214        }
215
216        if let Some(level) = self.get_level() {
217            let block_changed = !std::ptr::eq(old_block, new_block);
218            let moved_by_piston = flags.contains(UpdateFlags::UPDATE_MOVE_BY_PISTON);
219
220            if block_changed && (flags.contains(UpdateFlags::UPDATE_NEIGHBORS) || moved_by_piston) {
221                let behavior = REGISTRY.blocks.get_behavior(old_block);
222                behavior.affect_neighbors_after_removal(old_state, &*level, pos, moved_by_piston);
223            }
224
225            if !flags.contains(UpdateFlags::UPDATE_SKIP_ON_PLACE) {
226                let behavior = REGISTRY.blocks.get_behavior(new_block);
227                behavior.on_place(state, &*level, pos, old_state, moved_by_piston);
228            }
229        }
230
231        self.mark_unsaved();
232        Some(old_state)
233    }

235    /// Gets a block state at the given position.
236    #[must_use]
237    pub fn get_block_state(&self, pos: BlockPos) -> BlockStateId {
238        let y = pos.0.y;
239        let section_index = self.get_section_index(y);
240
241        // Bounds check - return air if out of range
242        if section_index >= self.sections.sections.len() {
243            return REGISTRY.blocks.get_base_state_id(vanilla_blocks::VOID_AIR);
244        }
245
246        let section = &self.sections.sections[section_index];
247        let section_guard = section.read();
248
249        if section_guard.states.has_only_air() {
250            return REGISTRY.blocks.get_base_state_id(vanilla_blocks::VOID_AIR);
251        }
252
253        let local_x = (pos.0.x & 15) as usize;
254        let local_y = (y & 15) as usize;
255        let local_z = (pos.0.z & 15) as usize;
256
257        section_guard.states.get(local_x, local_y, local_z)
258    }

260    /// Extracts the chunk data for sending to the client.
261    #[must_use]
262    pub fn extract_chunk_data(&self) -> ChunkPacketData {
263        let data = Vec::new();
264
265        let mut cursor = Cursor::new(data);
266        self.sections.sections.iter().for_each(|section| {
267            section.read().write(&mut cursor);
268        });
269
270        let heightmaps_guard = self.heightmaps.read();
271        ChunkPacketData {
272            heightmaps: Heightmaps {
273                heightmaps: vec![
274                    (
275                        ProtocolHeightmapType::WorldSurface,
276                        heightmaps_guard
277                            .get(HeightmapType::WorldSurface)
278                            .get_raw_data(),
279                    ),
280                    (
281                        ProtocolHeightmapType::MotionBlocking,
282                        heightmaps_guard
283                            .get(HeightmapType::MotionBlocking)
284                            .get_raw_data(),
285                    ),
286                    (
287                        ProtocolHeightmapType::MotionBlockingNoLeaves,
288                        heightmaps_guard
289                            .get(HeightmapType::MotionBlockingNoLeaves)
290                            .get_raw_data(),
291                    ),
292                ],
293            },
294            data: cursor.into_inner(),
295            block_entities: Vec::new(),
296        }
297    }

299    /// Extracts the light data for sending to the client.
300    #[must_use]
301    pub fn extract_light_data(&self) -> LightUpdatePacketData {
302        let section_count = self.sections.sections.len();
303        let mut sky_y_mask = BitSet(vec![0; section_count.div_ceil(64)].into_boxed_slice());
304        let mut block_y_mask = BitSet(vec![0; section_count.div_ceil(64)].into_boxed_slice());
305        let empty_sky_y_mask = BitSet(vec![0; section_count.div_ceil(64)].into_boxed_slice());
306        let empty_block_y_mask = BitSet(vec![0; section_count.div_ceil(64)].into_boxed_slice());
307
308        let mut sky_updates = Vec::new();
309        let mut block_updates = Vec::new();
310
311        for i in 0..section_count {
312            sky_y_mask.set(i, true);
313            block_y_mask.set(i, true);
314            sky_updates.push(vec![0xFF; 2048]);
315            block_updates.push(vec![0xFF; 2048]);
316        }
317
318        LightUpdatePacketData {
319            sky_y_mask,
320            block_y_mask,
321            empty_sky_y_mask,
322            empty_block_y_mask,
323            sky_updates,
324            block_updates,
325        }
326    }
